Triggers a message on entering age less than 18

DELIMITER //
CREATE TRIGGER DriverAgeChecker
BEFORE INSERT ON driver
FOR EACH ROW
BEGIN
IF TIMESTAMPDIFF(YEAR, NEW.date_of_birth, NOW()) < 18 THEN
SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'Aagya kal ka paida hua, 18 ka ho pehle.';
END IF;
END //
DELIMITER ;
