CREATE TRIGGER mobileNumberChecker
BEFORE INSERT ON d_contacts
FOR EACH ROW
BEGIN
DECLARE contact INT;
SET contact = LENGTH(NEW.contact);
IF contact <> 10 THEN
SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'asal number daal re kutreya';
END IF;
END //
